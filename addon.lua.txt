    Main.Name = "Main"
    Main.Size = UDim2.new(0, 450, 0, 330)
    Main.Position = UDim2.new(0.5, -225, 0.5, -165)
    Main.BackgroundColor3 = Color3.fromRGB(20, 18, 25)
    Main.ClipsDescendants = true
    Main.Active = true
    Main.Parent = ScreenGui

    AnimeBG.Size = UDim2.new(1, 0, 1, 0)
    AnimeBG.BackgroundTransparency = 1
    AnimeBG.Image = "rbxassetid://99396319452077"
    AnimeBG.ImageTransparency = 0
    AnimeBG.ScaleType = Enum.ScaleType.Crop
    AnimeBG.ZIndex = 1

    local BGCorner = Instance.new("UICorner", AnimeBG)

    Vignette.Name = "Vignette"
    Vignette.Size = UDim2.new(1, 0, 1, 0)
    Vignette.BackgroundTransparency = 1
    Vignette.Image = "rbxassetid://2576231974"
    Vignette.ImageTransparency = 0.4
    Vignette.ZIndex = 2

    local UIStroke = Instance.new("UIStroke", Main)
    UIStroke.Thickness = 2.5
    UIStroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border

    UIStroke.ZIndex = 4

            TweenService:Create(UIStroke, TweenInfo.new(2, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut), {Color = Color3.fromRGB(255, 0, 55), Transparency = 0.4}):Play()

    local TopBar = Instance.new("Frame", Main)

    TopBar.ZIndex = 5

    local Title = Instance.new("TextLabel", TopBar)
    Title.Size = UDim2.new(1, -20, 1, 0)
    Title.Position = UDim2.new(0, 15, 0, 0)
    Title.Text = "SURVIVE FROM MONSTER 💀"
    Title.TextColor3 = Color3.fromRGB(255, 255, 255)
    Title.Font = Enum.Font.GothamBold
    Title.TextSize = 16
    Title.TextXAlignment = Enum.TextXAlignment.Left
    Title.BackgroundTransparency = 1

    local Sidebar = Instance.new("ScrollingFrame", Main)

    Sidebar.Size = UDim2.new(0, 120, 1, -115)
    Sidebar.Position = UDim2.new(0, 12, 0, 45)
    Sidebar.BackgroundTransparency = 1
    Sidebar.CanvasSize = UDim2.new(0, 0, 0, 0)
    Sidebar.ScrollBarThickness = 0
    Sidebar.ZIndex = 5

    local SidebarLayout = Instance.new("UIListLayout", Sidebar)
    SidebarLayout.Padding = UDim.new(0, 6)

    local Container = Instance.new("Frame", Main)
    Container.Name = "Container"
    Container.Size = UDim2.new(1, -155, 1, -55)
    Container.Position = UDim2.new(0, 142, 0, 45)
    Container.BackgroundTransparency = 1
    Container.ZIndex = 5

    local ProfileFrame = Instance.new("Frame", Main)
    ProfileFrame.Size = UDim2.new(0, 120, 0, 50)
    ProfileFrame.Position = UDim2.new(0, 12, 1, -62)
    ProfileFrame.BackgroundColor3 = Color3.fromRGB(15, 12, 18)
    ProfileFrame.BackgroundTransparency = 0.5
    ProfileFrame.ZIndex = 5
    Instance.new("UICorner", ProfileFrame).CornerRadius = UDim.new(0, 8)
    
    local PStroke = Instance.new("UIStroke", ProfileFrame)
    PStroke.Thickness = 1.2
    PStroke.Color = Color3.fromRGB(255, 0, 55)
    PStroke.Transparency = 0.6

    local UserText = Instance.new("TextLabel", ProfileFrame)
    UserText.Size = UDim2.new(1, -10, 0, 16)
    UserText.Position = UDim2.new(0, 6, 0, 8)
    UserText.Text = "👤 " .. player.Name
    UserText.TextColor3 = Color3.fromRGB(255, 255, 255)
    UserText.Font = Enum.Font.GothamBold
    UserText.TextSize = 9
    UserText.TextXAlignment = Enum.TextXAlignment.Left
    UserText.BackgroundTransparency = 1
    UserText.ZIndex = 6

    local IdText = Instance.new("TextLabel", ProfileFrame)
    IdText.Size = UDim2.new(1, -10, 0, 14)
    IdText.Position = UDim2.new(0, 6, 0, 24)
    IdText.Text = "🆔 " .. player.UserId
    IdText.TextColor3 = Color3.fromRGB(180, 180, 180)
    IdText.Font = Enum.Font.GothamSemibold
    IdText.TextSize = 8
    IdText.TextXAlignment = Enum.TextXAlignment.Left
    IdText.BackgroundTransparency = 1
    IdText.ZIndex = 6

    local ToggleBtn = Instance.new("TextButton", ScreenGui)
    ToggleBtn.Size = UDim2.new(0, 95, 0, 32)
    ToggleBtn.Position = UDim2.new(0, 15, 0, 15)
    ToggleBtn.BackgroundColor3 = Color3.fromRGB(20, 15, 25)
    ToggleBtn.Text = "FIER TOGGLE"
    ToggleBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
    ToggleBtn.Font = Enum.Font.GothamBold
    ToggleBtn.TextSize = 10
    Instance.new("UICorner", ToggleBtn).CornerRadius = UDim.new(0, 6)
    
    local TStroke = Instance.new("UIStroke", ToggleBtn)
    TStroke.Thickness = 1.5
    TStroke.Color = Color3.fromRGB(160, 0, 255)

    local isOpened = true
    ToggleBtn.MouseButton1Click:Connect(function()
        isOpened = not isOpened
        local targetSize = isOpened and UDim2.new(0, 450, 0, 330) or UDim2.new(0, 450, 0, 0)
        local shadowSize = isOpened and UDim2.new(0, 470, 0, 350) or UDim2.new(0, 470, 0, 0)
        TweenService:Create(Main, TweenInfo.new(0.4, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {Size = targetSize}):Play()
        TweenService:Create(Shadow, TweenInfo.new(0.4, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {Size = shadowSize}):Play()
    end)

    local dragging, dragStart, startPos
    TopBar.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            dragging = true
            dragStart = input.Position
            startPos = Main.Position
            input.Changed:Connect(function()
                if input.UserInputState == Enum.UserInputState.End then dragging = false end
            end)
        end
    end)
    UserInputService.InputChanged:Connect(function(input)
        if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
            local delta = input.Position - dragStart
            local newPos = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
            local shadowPos = UDim2.new(startPos.X.Scale, (startPos.X.Offset - 10) + delta.X, startPos.Y.Scale, (startPos.Y.Offset - 10) + delta.Y)
            TweenService:Create(Main, TweenInfo.new(0.08, Enum.EasingStyle.Out), {Position = newPos}):Play()
            TweenService:Create(Shadow, TweenInfo.new(0.08, Enum.EasingStyle.Out), {Position = shadowPos}):Play()
        end
    end)

    local WindowObj = {CurrentTab = nil}

    function WindowObj:CreateTab(tabName)
        local TabPage = Instance.new("ScrollingFrame", Container)
        TabPage.Name = tabName .. "_Page"
        TabPage.Size = UDim2.new(1, 0, 1, 0)
        TabPage.BackgroundTransparency = 1
        TabPage.Visible = false
        TabPage.CanvasSize = UDim2.new(0, 0, 0, 0)
        TabPage.ScrollBarThickness = 2
        TabPage.ScrollBarImageColor3 = Color3.fromRGB(255, 0, 55)
        TabPage.ZIndex = 5

        local PageLayout = Instance.new("UIListLayout", TabPage)
        PageLayout.Padding = UDim.new(0, 7)
        PageLayout.SortOrder = Enum.SortOrder.LayoutOrder

        PageLayout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function()
            TabPage.CanvasSize = UDim2.new(0, 0, 0, PageLayout.AbsoluteContentSize.Y + 10)
        end)

        local TabBtn = Instance.new("TextButton", Sidebar)
        TabBtn.Name = tabName .. "_Btn"
        TabBtn.Size = UDim2.new(1, 0, 0, 32)
        TabBtn.BackgroundColor3 = Color3.fromRGB(30, 25, 35)
        TabBtn.BackgroundTransparency = 0.5
        TabBtn.Text = tabName
        TabBtn.TextColor3 = Color3.fromRGB(180, 180, 180)
        TabBtn.Font = Enum.Font.GothamBold
        TabBtn.TextSize = 11
        TabBtn.ZIndex = 6
        Instance.new("UICorner", TabBtn).CornerRadius = UDim.new(0, 6)

        local TabStroke = Instance.new("UIStroke", TabBtn)
        TabStroke.Thickness = 1
        TabStroke.Color = Color3.fromRGB(255, 0, 55)
        TabStroke.Enabled = false

        if WindowObj.CurrentTab == nil then
            WindowObj.CurrentTab = TabBtn
            TabBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
            TabBtn.BackgroundTransparency = 0.2
            TabStroke.Enabled = true
            TabPage.Visible = true
        end

        TabBtn.MouseButton1Click:Connect(function()
            if WindowObj.CurrentTab ~= TabBtn then
                for _, v in pairs(Container:GetChildren()) do
                    if v:IsA("ScrollingFrame") then v.Visible = false end
                end
                for _, v in pairs(Sidebar:GetChildren()) do
                    if v:IsA("TextButton") then
                        v.TextColor3 = Color3.fromRGB(180, 180, 180)
                        v.BackgroundTransparency = 0.5
                        if v:FindFirstChild("UIStroke") then v.UIStroke.Enabled = false end
                    end
                end
                WindowObj.CurrentTab = TabBtn
                TabBtn.TextColor3 = Color3.fromRGB(255, 255, 255)
                TabBtn.BackgroundTransparency = 0.2
                TabStroke.Enabled = true
                TabPage.Visible = true
            end
        end)

        local TabElements = {}

        function TabElements:CreateButton(btnText, callback)
            local callback = callback or function() end
            local Button = Instance.new("TextButton", TabPage)
            Button.Size = UDim2.new(1, -8, 0, 35)
            Button.BackgroundColor3 = Color3.fromRGB(25, 20, 30)
            Button.BackgroundTransparency = 0.4
            Button.Text = "  " .. btnText
            Button.TextColor3 = Color3.fromRGB(255, 255, 255)
            Button.Font = Enum.Font.GothamSemibold
            Button.TextSize = 12
            Button.TextXAlignment = Enum.TextXAlignment.Left
            Button.ZIndex = 6
            Instance.new("UICorner", Button).CornerRadius = UDim.new(0, 6)

            local BStroke = Instance.new("UIStroke", Button)
            BStroke.Thickness = 1
            BStroke.Color = Color3.fromRGB(60, 40, 70)

            Button.MouseButton1Click:Connect(function()
                pcall(callback)
                local oldColor = BStroke.Color
                BStroke.Color = Color3.fromRGB(255, 0, 55)
                task.wait(0.1)
                TweenService:Create(BStroke, TweenInfo.new(0.2), {Color = oldColor}):Play()
            end)
        end

        function TabElements:CreateToggle(toggleText, default, callback)
            local enabled = default or false
            local callback = callback or function() end

            local ToggleFrame = Instance.new("Frame", TabPage)
            ToggleFrame.Size = UDim2.new(1, -8, 0, 35)
            ToggleFrame.BackgroundColor3 = Color3.fromRGB(25, 20, 30)
            ToggleFrame.BackgroundTransparency = 0.4
            ToggleFrame.ZIndex = 6
            Instance.new("UICorner", ToggleFrame).CornerRadius = UDim.new(0, 6)

            local TStroke = Instance.new("UIStroke", ToggleFrame)
            TStroke.Thickness = 1
            TStroke.Color = Color3.fromRGB(60, 40, 70)

            local Label = Instance.new("TextLabel", ToggleFrame)
            Label.Size = UDim2.new(1, -50, 1, 0)
            Label.Position = UDim2.new(0, 10, 0, 0)
            Label.Text = toggleText
            Label.TextColor3 = Color3.fromRGB(255, 255, 255)
            Label.Font = Enum.Font.GothamSemibold
            Label.TextSize = 12
            Label.TextXAlignment = Enum.TextXAlignment.Left
            Label.BackgroundTransparency = 1
            Label.ZIndex = 7

            local Box = Instance.new("TextButton", ToggleFrame)
            Box.Size = UDim2.new(0, 36, 0, 20)
            Box.Position = UDim2.new(1, -46, 0.5, -10)
            Box.BackgroundColor3 = enabled and Color3.fromRGB(255, 0, 55) or Color3.fromRGB(40, 35, 45)
            Box.Text = ""
            Box.ZIndex = 7
            Instance.new("UICorner", Box).CornerRadius = UDim.new(0, 10)

            local Dot = Instance.new("Frame", Box)
            Dot.Size = UDim2.new(0, 14, 0, 14)
            Dot.Position = enabled and UDim2.new(1, -17, 0.5, -7) or UDim2.new(0, 3, 0.5, -7)
            Dot.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
            Dot.ZIndex = 8
            Instance.new("UICorner", Dot).CornerRadius = UDim.new(1, 0)

            Box.MouseButton1Click:Connect(function()
                enabled = not enabled
                local targetColor = enabled and Color3.fromRGB(255, 0, 55) or Color3.fromRGB(40, 35, 45)
                local targetPos = enabled and UDim2.new(1, -17, 0.5, -7) or UDim2.new(0, 3, 0.5, -7)
                
                TweenService:Create(Box, TweenInfo.new(0.2), {BackgroundColor3 = targetColor}):Play()
                TweenService:Create(Dot, TweenInfo.new(0.2, Enum.EasingStyle.Quart, Enum.EasingDirection.Out), {Position = targetPos}):Play()
                
                pcall(callback, enabled)
            end)
        end

        function TabElements:CreateSlider(sliderText, min, max, default, callback)
            local callback = callback or function() end
            local value = default or min

            local SliderFrame = Instance.new("Frame", TabPage)
            SliderFrame.Size = UDim2.new(1, -8, 0, 45)
            SliderFrame.BackgroundColor3 = Color3.fromRGB(25, 20, 30)
            SliderFrame.BackgroundTransparency = 0.4
            SliderFrame.ZIndex = 6
            Instance.new("UICorner", SliderFrame).CornerRadius = UDim.new(0, 6)

            local SStroke = Instance.new("UIStroke", SliderFrame)
            SStroke.Thickness = 1
            SStroke.Color = Color3.fromRGB(60, 40, 70)

            local Label = Instance.new("TextLabel", SliderFrame)
            Label.Size = UDim2.new(0.7, 0, 0, 25)
            Label.Position = UDim2.new(0, 10, 0, 2)
            Label.Text = sliderText
            Label.TextColor3 = Color3.fromRGB(255, 255, 255)
            Label.Font = Enum.Font.GothamSemibold
            Label.TextSize = 12
            Label.TextXAlignment = Enum.TextXAlignment.Left
            Label.BackgroundTransparency = 1
            Label.ZIndex = 7

            local ValLabel = Instance.new("TextLabel", SliderFrame)
            ValLabel.Size = UDim2.new(0.25, 0, 0, 25)
            ValLabel.Position = UDim2.new(1, -110, 0, 2)
            ValLabel.Text = tostring(value)
            ValLabel.TextColor3 = Color3.fromRGB(255, 0, 55)
            ValLabel.Font = Enum.Font.GothamBold
            ValLabel.TextSize = 12
            ValLabel.TextXAlignment = Enum.TextXAlignment.Right
            ValLabel.BackgroundTransparency = 1
            ValLabel.ZIndex = 7

            local Track = Instance.new("TextButton", SliderFrame)
            Track.Size = UDim2.new(1, -20, 0, 6)
            Track.Position = UDim2.new(0, 10, 1, -12)
            Track.BackgroundColor3 = Color3.fromRGB(45, 40, 50)
            Track.Text = ""
            Track.ZIndex = 7
            Instance.new("UICorner", Track).CornerRadius = UDim.new(1, 0)

            local Bar = Instance.new("Frame", Track)
            Bar.Size = UDim2.new((value - min) / (max - min), 0, 1, 0)
            Bar.BackgroundColor3 = Color3.fromRGB(255, 0, 55)
            Bar.ZIndex = 8
            Instance.new("UICorner", Bar).CornerRadius = UDim.new(1, 0)

            local function updateSlider(input)
                local percentage = math.clamp((input.Position.X - Track.AbsolutePosition.X) / Track.AbsoluteSize.X, 0, 1)
                value = math.floor(min + (max - min) * percentage)
                ValLabel.Text = tostring(value)
                Bar.Size = UDim2.new(percentage, 0, 1, 0)
                pcall(callback, value)
            end

            local sliding = false
            Track.InputBegan:Connect(function(input)
                if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
                    sliding = true
                    updateSlider(input)
                end
            end)
            UserInputService.InputChanged:Connect(function(input)
                if sliding and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
                    updateSlider(input)
                end
            end)
            UserInputService.InputEnded:Connect(function(input)
                if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
                    sliding = false
                end
            end)
        end

        return TabElements
    end

    return WindowObj
end

local Window = FierLib:CreateWindow()

local MainTab = Window:CreateTab("Main Script")
local PlayerTab = Window:CreateTab("Player Settings")

MainTab:CreateButton("Infinity Melon", function()
    local character = player.Character
    local hrp = character and character:FindFirstChild("HumanoidRootPart")
    if hrp then
        for _, obj in pairs(workspace:GetDescendants()) do
            if obj:IsA("TouchTransmitter") or (obj:IsA("ObjectValue") and obj.Name == "TouchInterest") then
                local parentPart = obj.Parent
                if parentPart and parentPart:IsA("BasePart") then
                    firetouchinterest(hrp, parentPart, 0)
                    firetouchinterest(hrp, parentPart, 1)
                end
            end
        end
    end
end)

MainTab:CreateButton("Bring All Melons", function()
    local character = player.Character
    local hrp = character and character:FindFirstChild("HumanoidRootPart")
    if hrp then
        for _, obj in pairs(workspace:GetDescendants()) do
            if obj.Name == "Melon" and obj:IsA("BasePart") then
                obj.CFrame = hrp.CFrame
            end
        end
    end
end)

MainTab:CreateButton("Buy Season Level", function()
    local controller = ReplicatedStorage:FindFirstChild("Controller")
    local summerForge = controller and controller:FindFirstChild("SummerForge")
    local event = summerForge and summerForge:FindFirstChild("buy")
    if event then
        event:FireServer(2)
    end
end)

MainTab:CreateButton("Buy Smoothies", function()
    local controller = ReplicatedStorage:FindFirstChild("Controller")
    local summerForge = controller and controller:FindFirstChild("SummerForge")
    local event = summerForge and summerForge:FindFirstChild("buy")
    if event then
        event:FireServer(1)
    end
end)

MainTab:CreateButton("Reset Server", function()
    local controller = ReplicatedStorage:FindFirstChild("Controller")
    local summerForge = controller and controller:FindFirstChild("SummerForge")

end)
